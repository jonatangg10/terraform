pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
                // Clonar el repositorio de Terraform
                git 'https://github.com/tu_usuario/tu_repositorio.git'
            }
        }

        stage('Terraform Init') {
            steps {
                // Inicializar Terraform
                sh 'terraform init'
            }
        }

        stage('Terraform Plan') {
            steps {
                // Planificar la infraestructura con Terraform
                sh 'terraform plan -out=tfplan'
            }
        }

        stage('Terraform Apply') {
            steps {
                // Aplicar los cambios de Terraform
                sh 'terraform apply -auto-approve tfplan'
            }
        }
    }

    post {
        success {
            // Notificar por correo electrónico si el pipeline se ejecuta con éxito
            emailext subject: "Terraform Pipeline Success",
                      body: "El pipeline de Terraform se ejecutó correctamente.",
                      to: "tu_correo@dominio.com"
        }
        failure {
            // Notificar por correo electrónico si el pipeline falla
            emailext subject: "Terraform Pipeline Failure",
                      body: "El pipeline de Terraform ha fallado. Por favor, revisa los registros para más detalles.",
                      to: "tu_correo@dominio.com"
        }
    }
}
